struct yen::TextAttributes
Struct used in Graphics Component to specify Text attributes.
        GraphicsComponent.h
Public Member Variables
unsigned int charactersSize

Size of characters displayed.
yen::Color textColor

Color of characters displayed.
Usage Example
        //Create Engine
yen::Engine engine;
EngineConfiguration configuration;
engine.initialize(configuration);

//Creating font
yen::FontManipulator fontManipulator = engine.resourceManager->addFontResource("my_font.otf");

//Creating object rendered as tekst
yen::ObjectManipulator tekstObject = engine.objectsManager->createObject(yen::fVector(0.0f, 0.0f));
yen::GraphicsComponent tekstGraphicsComponent(engine.graphicsEngine, yen::GraphicsComponentContent::TEXT);
yen::ComponentManipulator tekstGraphicsComponentManipulator;
engine.objectsManager->attachComponent(&tekstGraphicsComponentManipulator, tekstObject, &tekstGraphicsComponent); 

//Creating and using TextAttributes
yen::TextAttributes tAttr;
tAttr.charactersSize = 12;
tAttr.textColor = yen::Color(0, 0, 0, 255);
textGComp.setFontResource(fontManipulator);
textGComp.setString(sManipulator, "car");
textGComp.setTextAttributes(tAttr);

//Create scene and add Camera object to it
yen::SceneManipulator scene = engine.sceneManager->createScene();
engine.sceneManager->addObjectToScene(scene, tekstObject);
